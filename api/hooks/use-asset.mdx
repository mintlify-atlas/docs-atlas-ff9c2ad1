---
title: "useAsset"
description: "React hook for managing asset state and access details"
---

The `useAsset` hook provides comprehensive asset management including fetching, access control, and service credential verification.

## Hook Signature

```tsx
const useAsset = (): AssetProviderValue
```

## Usage

The hook must be used within an `AssetProvider`:

```tsx
import { AssetProvider, useAsset } from '@context/Asset'

function AssetWrapper({ did, children }) {
  return (
    <AssetProvider did={did}>
      {children}
    </AssetProvider>
  )
}

function AssetDetails() {
  const { asset, loading, error, isOwner, fetchAsset } = useAsset()
  
  if (loading) return <Loader />
  if (error) return <Error message={error} />
  
  return (
    <div>
      <h1>{asset.metadata.name}</h1>
      {isOwner && <EditButton />}
    </div>
  )
}
```

## Features

### Automatic Asset Fetching

The hook automatically fetches asset data when the DID changes:

```tsx
useEffect(() => {
  if (!isMounted || !appConfig?.metadataCacheUri) return
  fetchAsset(newCancelToken())
}, [appConfig?.metadataCacheUri, fetchAsset, newCancelToken, isMounted])
```

### Access Details

Automatically fetches and updates access details:

```tsx
const accessDetails = await getAccessDetails(
  asset.chainId,
  asset.services[0].datatokenAddress,
  asset.services[0].timeout,
  accountId
)
```

### Service Credential Verification

Verifies Gaia-X service credentials if present:

```tsx
const { verified, complianceApiVersion, idMatch, isIdMatchVerifiable } =
  await verifyRawServiceCredential(serviceCredentialContent, asset.id)
```

### Network and Ownership Checks

```tsx
// Check if on correct network
const isAssetNetwork = chain?.id === asset?.chainId

// Check if user is owner
const isOwner = accountId?.toLowerCase() === owner.toLowerCase()
```

## Example: Complete Asset Page

```tsx
import { AssetProvider, useAsset } from '@context/Asset'
import { useAccount } from 'wagmi'

function AssetPage({ did }: { did: string }) {
  return (
    <AssetProvider did={did}>
      <AssetContent />
    </AssetProvider>
  )
}

function AssetContent() {
  const { address } = useAccount()
  const {
    asset,
    loading,
    error,
    isOwner,
    isAssetNetwork,
    isInPurgatory,
    assetState
  } = useAsset()
  
  if (loading) return <div>Loading asset...</div>
  if (error) return <div>Error: {error}</div>
  if (isInPurgatory) return <div>Asset is in purgatory</div>
  
  return (
    <div>
      <h1>{asset.metadata.name}</h1>
      <p>State: {assetState}</p>
      <p>Owner: {isOwner ? 'You' : asset.nft.owner}</p>
      
      {!isAssetNetwork && (
        <Alert>Please switch to the correct network</Alert>
      )}
      
      {isOwner && <EditAssetButton />}
      <AssetActions asset={asset} />
    </div>
  )
}
```
